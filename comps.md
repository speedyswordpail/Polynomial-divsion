public class comp
{
   private int pow;
   private int coef;
   private boolean pn;
   public comp(int power,int coeff,boolean posneg)
   {
      power = pow;
      coeff = coef;
      posneg = pn;
   }
   
   public int getPower()
   {
      return pow;
   }
   public int getCoeff()
   {
      return coef;
   }
   public boolean getSign()
   {
      return pn;
   }
}
